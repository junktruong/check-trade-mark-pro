API: POST /api/trademark/status

Case: SAFE (no warning, no block, no TMHunt matches)
{
  "status": "safe",
  "warningWords": [],
  "blockedWords": [],
  "tmhuntWords": [],
  "ai_status": "SAFE",
  "ai_reason": ""
}

Case: WARNING (warning words found, no block)
{
  "status": "warning",
  "warningWords": ["limited", "pro"],
  "blockedWords": [],
  "tmhuntWords": [],
  "ai_status": "WARNING",
  "ai_reason": "Dùng từ cấm Perfect"
}

Case: BLOCK (blocklist hit)
{
  "status": "block",
  "warningWords": [],
  "blockedWords": ["nintendo"],
  "tmhuntWords": [],
  "ai_status": "SAFE",
  "ai_reason": ""
}

Case: BLOCK (TMHunt live mark)
{
  "status": "block",
  "warningWords": [],
  "blockedWords": ["starbucks"],
  "tmhuntWords": ["starbucks"],
  "ai_status": "SAFE",
  "ai_reason": ""
}

Case: BLOCK (blocklist + TMHunt)
{
  "status": "block",
  "warningWords": ["pro"],
  "blockedWords": ["apple", "coca-cola"],
  "tmhuntWords": ["coca-cola"],
  "ai_status": "WARNING",
  "ai_reason": "Nhồi nhét từ khóa"
}

Case: ERROR (HTTP 500)
{
  "error": "TMHunt HTTP 500"
}
